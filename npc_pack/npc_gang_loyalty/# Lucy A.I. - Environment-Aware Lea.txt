def respond(text):
    if CONFIG.get("text_feedback"):
        print(f"Lucy: {text}")
    if CONFIG.get("voice_feedback"):
        engine.say(text)
        engine.runAndWait()
    if CONFIG.get("popup_feedback") and CONFIG.get("ui_enabled"):
        messagebox.showinfo("Lucy", text)

def ensure_directories():
    os.makedirs(CONFIG['memory_crate_dir'], exist_ok=True)

def load_memory():
    if os.path.exists(CONFIG['memory_file']):
        with open(CONFIG['memory_file'], 'r') as f:
            global lucy_memory
            lucy_memory = json.load(f)

def save_memory():
    with open(CONFIG['memory_file'], 'w') as f:
        json.dump(lucy_memory, f, indent=4)

def save_crate(crate_name, data):
    crate_path = os.path.join(CONFIG['memory_crate_dir'], f"{crate_name}.json")
    os.makedirs(os.path.dirname(crate_path), exist_ok=True)
    with open(crate_path, 'w') as f:
        json.dump(data, f, indent=4)
    log(f"Saved memory crate: {crate_name}")

def organize_by_context(path):
    if CONFIG.get("organization_by_context"):
        context = "fxserver" if "FXServer" in path else "txadmin" if "txAdmin" in path else "documents"
        if context not in lucy_memory["contextual_folders"]:
            lucy_memory["contextual_folders"][context] = []
        if path not in lucy_memory["contextual_folders"][context]:
            lucy_memory["contextual_folders"][context].append(path)
            log(f"Organized {path} under context: {context}")

def log(message):
    with open(CONFIG['log_file'], 'a') as f:
        f.write(f"[{datetime.now()}] {message}\n")

def run_shell(cmd):
    if CONFIG.get("allow_shell_commands"):
        try:
            output = subprocess.check_output(cmd, shell=True, stderr=subprocess.STDOUT, text=True)
            log(f"Shell command output: {output}")
            respond(f"Executed: {cmd}")
            return output
        except subprocess.CalledProcessError as e:
            log(f"Shell error: {e.output}")
            respond("Shell command failed")

def open_program_by_name(name):
    programs = CONFIG.get("trusted_programs", [])
    if name.lower() in programs:
        run_exe(f"{name}.exe")
    else:
        respond("Program not trusted or unknown")

def close_program_by_name(name):
    programs = CONFIG.get("trusted_programs", [])
    if name.lower() in programs:
        try:
            subprocess.call(["taskkill", "/f", "/im", f"{name}.exe"], shell=True)
            respond(f"Closed {name} successfully")
            log(f"Closed program: {name}")
        except Exception as e:
            log(f"Failed to close {name}: {str(e)}")
            respond("Failed to close the program")
    else:
        respond("Program not trusted or unknown")

def simulate_keypress(keys):
    respond(f"Simulating: {keys}")
    pyautogui.hotkey(*keys.split('+'))

def clone_github_repo(url, destination="lucy_clones"):
    if CONFIG.get("auto_clone_github"):
        os.makedirs(destination, exist_ok=True)
        cmd = f"git clone {url} {destination}"
        run_shell(cmd)
